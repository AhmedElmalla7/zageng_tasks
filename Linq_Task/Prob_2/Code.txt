
record Product(int Id, string Name, decimal Price, string Category);

namespace Problem2
{
    internal class Program
    {
        static void Main(string[] args)
        {
            List<Product> products = 
                [
                new(1,"Laptop", 1200m,"Electronics"),
                new(2,"Phone", 800m,"Electronics"),
                new(3,"Desk", 350m,"Furniture"),
                new(4,"Chair", 150m,"Furniture"),
                new(5,"Headphones", 200m,"Electronics") 
                ];

            // 1. Get the first Electronics product
            Console.WriteLine("              First / First or default");
            // 1_1 . First : throws an exception if not exist
            try
            {
                var firstElecProduct_1 = products.First(p => p.Category == "Electronics"); 
                Console.WriteLine($"First Electronic Product : {firstElecProduct_1.Name}");
            }
            catch (InvalidOperationException)
            {
                Console.WriteLine("No Electronic products exist");
            }
            // 1_2 . First or default : returns null if not exist
            var firstElecProduct_2 = products.FirstOrDefault(p => p.Category == "Electronics"); 
            if (firstElecProduct_2 == null)
            {
                Console.WriteLine("No products exist");
            }
            else
            {
                Console.WriteLine($"First Electronic Product : {firstElecProduct_2.Name}");
            }
            Console.WriteLine("--------------------------------------------");

            // 2. Get the last product with Price > 1000 (use OrDefault — handle null)
            Console.WriteLine("              Last / Last or default");
            // 2_1 . Last : throws an exception if not exist
            try
            {
                var lastProductGreaterThan1000_1 = products.Last(p => p.Price > 1000); 
                Console.WriteLine($"Last Product greater than 1000 : {lastProductGreaterThan1000_1.Name} ( price : {lastProductGreaterThan1000_1.Price} )");
            }
            catch (InvalidOperationException)
            {
                Console.WriteLine("No products exist greater than 1000");
            }
            // 2_2 . Last or default : returns null if not exist
            var lastProductGreaterThan1000_2 = products.LastOrDefault(p => p.Price > 1000); 
            if(lastProductGreaterThan1000_2 == null)
            {
                Console.WriteLine("No products exist greater than 1000");
            }
            else
            {
                Console.WriteLine($"Last Product greater than 1000 : {lastProductGreaterThan1000_2.Name} ( price : {lastProductGreaterThan1000_2.Price} )");
            }
            Console.WriteLine("--------------------------------------------");

            // 3.Get the single Furniture item with Price > 300(what if > 1 match ?)
            Console.WriteLine("              Single / Single or default");
            // single : throws an exception if not found or duplication
            try
            {
                var singleFurnGreaterThan300_1 = products.Single(p => p.Price > 300 && p.Category == "Furniture");
                Console.WriteLine($"single Furniture Greater Than 300 : {singleFurnGreaterThan300_1.Name} ( price : {singleFurnGreaterThan300_1.Price} ) ");
            }
            catch (Exception e)
            {
                Console.WriteLine(e.Message);        
            }
            // Single or default : returns null if not found but throw exception if there is a duplication
            try
            {
                var singleFurnGreaterThan300_2 = products.SingleOrDefault(p => p.Price > 300 && p.Category == "Furniture");
                Console.WriteLine($"single Furniture Greater Than 300 : {(singleFurnGreaterThan300_2 is null ? "null" : singleFurnGreaterThan300_2.Name + " ( price : " + singleFurnGreaterThan300_2.Price +" )")}");

            }
            catch (Exception e)
            {
                Console.WriteLine(e.Message);
            }
            Console.WriteLine("--------------------------------------------");
            // 4. Get the element at index 3
            Console.WriteLine("              Element at / Element at or default");
            // element at : throws an exception if index is out of range
            try
            {
                var elementAtIndex3_1 = products.ElementAt(3);
                Console.WriteLine($"The Element at index 3 : {elementAtIndex3_1.Name}");
            }
            catch (ArgumentOutOfRangeException)
            {
                Console.WriteLine("index is out of range");
            }
            catch (Exception e)
            {
                Console.WriteLine(e.Message);
            }
            // element at or default : returns null if not found
            var elementAtIndex3_2 = products.ElementAtOrDefault(3);
            Console.WriteLine($"The Element at index 3 : {(elementAtIndex3_2 is null ? "null" :elementAtIndex3_2.Name)}");
        }
    }
}
